27 class SimplePI {

29  struct Config {

32 
33  template <typename Archive>
34  void Serialize(Archive* a) {

40  struct State {
41  float integral = 0.0f;
42  // When starting with desired rate limits in place, we by default
43  // always accept the first desired command with no limiting (users
44  // can of course override this value if they want to start from
45  // some predetermined value).
46  float desired = std::numeric_limits<float>::quiet_NaN();
47 
48  // The following are not actually part of the "state", but are
49  // present for purposes of being logged with it.
50  float error = 0.0f;
51 
52  float p = 0.0f;
53  float command = 0.0f;
54 
55  void Clear() MOTEUS_CCM_ATTRIBUTE {
56  // We implement this solely for speed, because on at least
57  // Cortex-M4, just calling the constructor delegates to memset,
58  // which is much slower than memberwise assignment.
59  integral = 0.0f;
60 
61  desired = std::numeric_limits<float>::quiet_NaN();
62 
63  error = 0.0f;
64  p = 0.0f;
65  command = 0.0f;
66  }
67 
68  template <typename Archive>
69  void Serialize(Archive* a) {
70  a->Visit(MJ_NVP(integral));
71  a->Visit(MJ_NVP(desired));
72  a->Visit(MJ_NVP(error));
73  a->Visit(MJ_NVP(p));
74  a->Visit(MJ_NVP(command));
75  }
76  };
77 
78  SimplePI(const Config* config, State* state)
79  : config_(config), state_(state) {}
80 
81  float Apply(float measured, float input_desired,
82  int rate_hz) MOTEUS_CCM_ATTRIBUTE {
83  float desired = {};
84 
85  desired = input_desired;
86 
87  state_->desired = desired;
88  state_->error = measured - desired;
89 
90  float to_update_i = state_->error * config_->ki / rate_hz;
91  state_->integral += to_update_i;
92 
93  state_->p = config_->kp * state_->error;
94  state_->command = -1.0f * (state_->p + state_->integral);
95 
96  return state_->command;
97  }
98 
99  private:
100  const Config* const config_;
101  State* const state_;
102 };

104 }
