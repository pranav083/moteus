32  struct Options {
33  PinName mosi = NC;
34  PinName miso = NC;
35  PinName sck = NC;
36  PinName cs = NC;
37  int frequency = 10000000;
38  int width = 16;
39  int mode = 1;
40  uint16_t timeout = 20000;

42  // Only necessary if DMA operations will be used.
43  DMA_Channel_TypeDef* rx_dma = nullptr;

45  };

47  Stm32Spi(const Options& options)
48  : cs_(std::in_place_t(), options.cs, 1),
49  options_(options) {
50 
51  spi_init(&spi_, options.mosi, options.miso, options.sck, NC);
52  spi_format(&spi_, options.width, options.mode, 0);
53  spi_frequency(&spi_, options.frequency);
54 
55  auto* const spi = spi_.spi.handle.Instance;
56  spi->CR1 &= ~SPI_CR1_SPE;
57 
58  if (options_.rx_dma ||
59  options_.tx_dma) {
60  MJ_ASSERT(options_.rx_dma);
61  MJ_ASSERT(options_.tx_dma);
62 
63  __HAL_RCC_DMAMUX1_CLK_ENABLE();
64  __HAL_RCC_DMA1_CLK_ENABLE();
65  __HAL_RCC_DMA2_CLK_ENABLE();
66 
67  dmamux_rx_ = Stm32Dma::SelectDmamux(options_.rx_dma);
68  dmamux_tx_ = Stm32Dma::SelectDmamux(options_.tx_dma);
69 
70  options_.rx_dma->CCR =
71  DMA_PERIPH_TO_MEMORY |
72  DMA_PINC_DISABLE |
73  DMA_MINC_ENABLE |
74  DMA_PDATAALIGN_BYTE |
75  DMA_MDATAALIGN_BYTE |
76  DMA_PRIORITY_HIGH;
77  options_.tx_dma->CCR =
78  DMA_MEMORY_TO_PERIPH |
79  DMA_PINC_DISABLE |
80  DMA_MINC_ENABLE |
81  DMA_PDATAALIGN_BYTE |
82  DMA_MDATAALIGN_BYTE |
83  DMA_PRIORITY_HIGH;
84  dmamux_rx_->CCR = GetSpiRxRequest(spi) & DMAMUX_CxCR_DMAREQ_ID;
85  dmamux_tx_->CCR = GetSpiTxRequest(spi) & DMAMUX_CxCR_DMAREQ_ID;
86 
87  options_.tx_dma->CPAR = u32(&spi->DR);
88  options_.rx_dma->CPAR = u32(&spi->DR);
89  }
90  }

92  void set_cs(PinName cs) {
93  // The SPI class may be used from an interrupt. If we want to
94  // change the CS line, we have to make sure no one can use it
95  // while it is being changed.
96  __disable_irq();
97  cs_.emplace(cs, 1);
98  __enable_irq();
99  }

101  uint16_t write(uint16_t value) MOTEUS_CCM_ATTRIBUTE {
102  start_write(value);
103  return finish_write();
104  }
105 
106  void start_write(uint16_t value) MOTEUS_CCM_ATTRIBUTE {
107  auto* const spi = spi_.spi.handle.Instance;
108  *cs_ = 0;
109 
110  // This doesn't seem to be a whole lot faster than the HAL in
111  // practice, but it doesn't hurt to do it ourselves and not have
112  // to worry about the extra stuff the HAL does.
113  uint16_t timeout = options_.timeout;
114  while (((spi->SR & SPI_SR_BSY) != 0) && timeout) { timeout--; }
115  spi->DR = value;
116  spi->CR1 |= SPI_CR1_SPE;
117  }
118 
119  uint16_t finish_write() MOTEUS_CCM_ATTRIBUTE {
120  auto* const spi = spi_.spi.handle.Instance;
121 
122  uint16_t timeout = options_.timeout;
123 
124  while (((spi->SR & SPI_SR_RXNE) == 0) && timeout) { timeout--; }
125  const uint16_t result = spi->DR;
126  while (((spi->SR & SPI_SR_TXE) == 0) && timeout) { timeout--; }
127  while (((spi->SR & SPI_SR_BSY) != 0) && timeout) { timeout--; }
128  spi->CR1 &= ~(SPI_CR1_SPE);
129 
130  *cs_ = 1;
131  return result;
132  }

134  void start_dma_transfer(
135  std::string_view tx_buffer,
136  mjlib::base::string_span rx_buffer) MOTEUS_CCM_ATTRIBUTE {
137  *cs_ = 0;
138 
139  auto* const spi = spi_.spi.handle.Instance;
140 
141  // Empty out the receive FIFO.
142  while (spi->SR & SPI_SR_FRLVL_Msk) {
143  (void)spi->DR;
144  }
145 
146  // We should not have a transaction operating at the moment.
147  // MJ_ASSERT((spi->CR2 & (SPI_CR2_TXDMAEN | SPI_CR2_RXDMAEN)) == 0);
148  // MJ_ASSERT(tx_buffer.size() == static_cast<size_t>(rx_buffer.size()));
149 
150  options_.rx_dma->CNDTR = tx_buffer.size();
151  options_.tx_dma->CNDTR = rx_buffer.size();
152 
153  options_.rx_dma->CMAR = u32(&rx_buffer[0]);
154  options_.tx_dma->CMAR = u32(&tx_buffer[0]);
155 
156  spi->CR2 |= SPI_CR2_RXDMAEN;
157 
158  options_.tx_dma->CCR |= DMA_CCR_EN;
159  options_.rx_dma->CCR |= DMA_CCR_EN;
160 
161  spi->CR2 |= SPI_CR2_TXDMAEN;
162 
163  spi->CR1 |= SPI_CR1_SPE;
164  }
165 
166  bool is_dma_finished() MOTEUS_CCM_ATTRIBUTE {
167  auto* const spi = spi_.spi.handle.Instance;
168 
169  return
170  ((spi->SR & SPI_SR_BSY) == 0) &&
171  ((spi->SR & SPI_SR_FTLVL_Msk) == 0) &&
172  (options_.tx_dma->CNDTR == 0) &&
173  (options_.rx_dma->CNDTR == 0);
174  }
175 
176  void finish_dma_transfer() MOTEUS_CCM_ATTRIBUTE {
177  auto* const spi = spi_.spi.handle.Instance;
178 
179  while (!is_dma_finished());
180 
181  options_.rx_dma->CCR &= ~(DMA_CCR_EN);
182  options_.tx_dma->CCR &= ~(DMA_CCR_EN);
183 
184  spi->CR1 &= ~(SPI_CR1_SPE);
185 
186  spi->CR2 &= ~(SPI_CR2_TXDMAEN | SPI_CR2_RXDMAEN);
187 
188  *cs_ = 1;
189  }

191  private:
192  static uint32_t GetSpiTxRequest(SPI_TypeDef* spi) {
193  switch (u32(spi)) {
194  case SPI_1: return DMA_REQUEST_SPI1_TX;
195  case SPI_2: return DMA_REQUEST_SPI2_TX;
196  case SPI_3: return DMA_REQUEST_SPI3_TX;
197  }
198  mbed_die();
199  }
200 
201  static uint32_t GetSpiRxRequest(SPI_TypeDef* spi) {
202  switch (u32(spi)) {
203  case SPI_1: return DMA_REQUEST_SPI1_RX;
204  case SPI_2: return DMA_REQUEST_SPI2_RX;
205  case SPI_3: return DMA_REQUEST_SPI3_RX;
206  }
207  mbed_die();
208  }
