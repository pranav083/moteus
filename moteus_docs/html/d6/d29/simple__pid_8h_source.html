46  a->Visit(MJ_NVP(sign));

50  struct State {
51  float integral = 0.0f;
52  // When starting with desired rate limits in place, we by default
53  // always accept the first desired command with no limiting (users
54  // can of course override this value if they want to start from
55  // some predetermined value).
56  float desired = std::numeric_limits<float>::quiet_NaN();
57 
58  // The following are not actually part of the "state", but are
59  // present for purposes of being logged with it.
60  float error = 0.0f;
61  float error_rate = 0.0f;
62 
63  float p = 0.0f;
64  float d = 0.0f;
65  float pd = 0.0f;
66  float command = 0.0f;
67 
68  void Clear() MOTEUS_CCM_ATTRIBUTE {
69  // We implement this solely for speed, because on at least
70  // Cortex-M4, just calling the constructor delegates to memset,
71  // which is much slower than memberwise assignment.
72  integral = 0.0f;
73 
74  desired = std::numeric_limits<float>::quiet_NaN();
75 
76  error = 0.0f;
77  error_rate = 0.0f;
78  p = 0.0f;
79  d = 0.0f;
80  pd = 0.0f;
81  command = 0.0f;
82  }
83 
84  template <typename Archive>
85  void Serialize(Archive* a) {
86  a->Visit(MJ_NVP(integral));
87  a->Visit(MJ_NVP(desired));
88  a->Visit(MJ_NVP(error));
89  a->Visit(MJ_NVP(error_rate));
90  a->Visit(MJ_NVP(p));
91  a->Visit(MJ_NVP(d));
92  a->Visit(MJ_NVP(pd));
93  a->Visit(MJ_NVP(command));

97  PID(const Config* config, State* state)
98  : config_(config), state_(state) {}

100  struct ApplyOptions {
101  float kp_scale = 1.0f;
102  float kd_scale = 1.0f;
103 
104  ApplyOptions() {}
105  };
106 
107  float Apply(float measured, float input_desired,
108  float measured_rate, float input_desired_rate,
109  int rate_hz,
110  ApplyOptions apply_options = {}) MOTEUS_CCM_ATTRIBUTE {
111  float desired = {};
112  float desired_rate = {};
113 
114  // First apply max_desired_rate
115  if (config_->max_desired_rate != 0.0f &&
116  std::isfinite(state_->desired)) {
117  const float max_step = config_->max_desired_rate / rate_hz;
118  const float proposed_step = input_desired - state_->desired;
119  const float actual_step =
120  mjlib::base::Limit(proposed_step, -max_step, max_step);
121  desired = state_->desired + actual_step;
122  desired_rate =
123  mjlib::base::Limit(input_desired_rate, -config_->max_desired_rate,
124  config_->max_desired_rate);
125  } else {
126  desired = input_desired;
127  desired_rate = input_desired_rate;
128  }
129 
130  state_->desired = desired;
131  state_->error = measured - desired;
132  state_->error_rate = measured_rate - desired_rate;
133 
134  const float max_i_update = config_->iratelimit / rate_hz;
135  float to_update_i = state_->error * config_->ki / rate_hz;
136  if (max_i_update > 0.0f) {
137  if (to_update_i > max_i_update) {
138  to_update_i = max_i_update;
139  } else if (to_update_i < -max_i_update) {
140  to_update_i = -max_i_update;
141  }
142  }
143 
144  state_->integral += to_update_i;
145 
146  if (state_->integral > config_->ilimit) {
147  state_->integral = config_->ilimit;
148  } else if (state_->integral < -config_->ilimit) {
149  state_->integral = -config_->ilimit;
150  }
151 
152  state_->p = apply_options.kp_scale * config_->kp * state_->error;
153  state_->d = apply_options.kd_scale * config_->kd * state_->error_rate;
154  state_->pd = state_->p + state_->d;
155 
156  state_->command = config_->sign * (state_->pd + state_->integral);
157 
158  return state_->command;
159  }
160 
161  private:
162  const Config* const config_;
163  State* const state_;
164 };
