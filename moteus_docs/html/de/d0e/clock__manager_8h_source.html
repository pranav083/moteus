27 class ClockManager {
28  public:
29  static constexpr int kMaxExtraTrim = 8;
30 
31  ClockManager(MillisecondTimer* timer,
32  mjlib::micro::PersistentConfig& persistent_config,
33  mjlib::micro::CommandManager& command_manager)
34  : timer_(timer) {
35  persistent_config.Register("clock", &clock_, [this]() {
36  this->UpdateConfig();
37  });
38  command_manager.Register("clock", [this](auto&& command, auto&& response) {
39  this->Command(command, response);
40  });
41  }
42 
43  void UpdateConfig() {
44  const int32_t trim =
45  std::max<int32_t>(
46  0, std::min<int32_t>(127, clock_.hsitrim + extra_trim_));
47  RCC->ICSCR = (RCC->ICSCR & ~0xff000000) | (trim << 24);
48  }
49 
50  void SetTrim(int extra_trim) {
51  extra_trim_ = std::max(-kMaxExtraTrim, std::min(kMaxExtraTrim, extra_trim));
52  UpdateConfig();
53  }
54 
55  int trim() const {
56  return extra_trim_;
57  }
58 
59  void Command(const std::string_view& command,
60  const mjlib::micro::CommandManager::Response& response) {
61  mjlib::base::Tokenizer tokenizer(command, " ");
62  const auto cmd_text = tokenizer.next();
63 
64  if (cmd_text == "us") {
65  snprintf(output_, sizeof(output_), "%" PRIu32 "\r\n",
66  static_cast<uint32_t>(timer_->read_us()));
67  WriteMessage(output_, response);
68  } else if (cmd_text == "trim") {
69  const auto value_str = tokenizer.next();
70 
71  if (value_str == "") {
72  snprintf(output_, sizeof(output_), "%d\r\n", extra_trim_);
73  WriteMessage(output_, response);
74  } else {
75  SetTrim(std::strtol(value_str.data(), nullptr, 0));
76  WriteMessage("OK\r\n", response);
77  }
78  } else {
79  WriteMessage("ERR unknown clock\r\n", response);
80  }
81  }
82 
83  void WriteMessage(const std::string_view& message,
84  const mjlib::micro::CommandManager::Response& response) {
85  mjlib::micro::AsyncWrite(*response.stream, message, response.callback);
86  }

96  };
