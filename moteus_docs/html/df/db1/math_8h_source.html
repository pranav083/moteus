33 inline float WrapZeroToTwoPi(float x) {
34  // We would use fmodf, but we're trying to be fast here and don't
35  // care too much about accuracy.
36 
37  // const float mod = ::fmodf(x, k2Pi);
38 
39  const int32_t divisor = static_cast<int>(x / k2Pi);
40  const float mod = x - divisor * k2Pi;
41  return (mod >= 0.0f) ? mod : (mod + k2Pi);
42 }

44 int32_t RadiansToQ31(float) __attribute__((always_inline));
45 
46 inline int32_t RadiansToQ31(float x) {
47  // First we scale, then wrap, and finally convert out.
48  const float scaled = x / k2Pi;
49  // Now we wrap to be from 0 to 1.
50  const int32_t i = static_cast<int>(scaled);
51  float mod = scaled - i;
52  if (mod < 0) { mod += 1.0f; }
53 
54  return static_cast<int32_t>(((mod > 0.5f) ? (mod - 1.0f) : mod) * 4294967296.0f);
55 }

64 // We re-implement frexpf so that it can get stuck into the CCM RAM.
65 // Otherwise it calls out to the libc one, which will be much slower
66 // since it is in normal flash.
67 float my_frexpf(float x, int* pw2) __attribute((always_inline));
68 
69 inline float my_frexpf(float x, int *pw2) {
70  detail::float_long fl;
71  long int i;
72 
73  fl.f = x;
74  i = (fl.l >> 23) & 0x000000ff;
75  i -= 0x7e;
76  *pw2 = i;
77  fl.l &= 0x807fffff;
78  fl.l |= 0x3f000000;
79  return fl.f;
80 }

84 inline float log2f_approx(float X) {
85  // From: http://openaudio.blogspot.com/2017/02/faster-log10-and-pow.html
86  float Y, F;
87  int E;
88  F = my_frexpf(fabsf(X), &E);
89  Y = 1.23149591368684f;
90  Y *= F;
91  Y += -4.11852516267426f;
92  Y *= F;
93  Y += 6.02197014179219f;
94  Y *= F;
95  Y += -3.13396450166353f;
96  Y += E;
97  return X == 1.0f ? 0.0f : Y;
98 }

102 inline float pow2f_approx(float x) {
103  // From: https://gist.github.com/petrsm/079de9396d63e00d5994a7cc936ae9c7
104 
105  volatile union {
106  float f;
107  unsigned int i;
108  } cvt;
109 
110  const float pi = static_cast<int>(x);
111  const float pf = x - pi;
112  cvt.i = (1 << 23) * (static_cast<int>(x) + 127);
113  const float pow2i = cvt.f;
114  float pow2f = 7.9204240219773237e-2f;
115 
116  pow2f = pow2f * pf + 2.2433836478672357e-1f;
117  pow2f = pow2f * pf + 6.9645739499350319e-1f;
118  pow2f = pow2f * pf + 9.9989296565051542e-1f;
119 
120  return pow2i * pow2f;
121 }

123 inline float FastAtan2(float y, float x) {
124  // Based on https://math.stackexchange.com/a/1105038/81278
125 
126  // a := min (|x|, |y|) / max (|x|, |y|)
127  const float abs_y = std::abs(y);
128  const float abs_x = std::abs(x);
129  // inject FLT_MIN in denominator to avoid division by zero
130  const float a = std::min(abs_x, abs_y) / (std::max(abs_x, abs_y) + std::numeric_limits<float>::min());
131  // s := a * a
132  const float s = a * a;
133  // r := ((-0.0464964749 * s + 0.15931422) * s - 0.327622764) * s * a + a
134  float r = ((-0.0464964749f * s + 0.15931422f) * s - 0.327622764f) * s * a + a;
135  // if |y| > |x| then r := 1.57079637 - r
136  if (abs_y > abs_x) {
137  r = 1.57079637f - r;
138  }
139  // if x < 0 then r := 3.14159274 - r
140  if (x < 0.0f) {
141  r = 3.14159274f - r;
142  }
143  // if y < 0 then r := -r
144  if (y < 0.0f) {
145  r = -r;
146  }
147 
148  return r;
149 }
