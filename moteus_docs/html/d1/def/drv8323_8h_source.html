180 
181  // Gate Drive HS Register
182 
183  // hw rev 7 boards use a drv8353, which is sensitive to damage
184  // ringing on the gate drives. This version requires lower gate
185  // drive strength to avoid damage. hw rev 8 boards have a better
186  // layout and an additional gate drive resistor which allows them
187  // to go higher.
188  uint16_t idrivep_hs_ma =
189  g_measured_hw_family == 0 ?
190  ((g_measured_hw_rev <= 6) ? 370 :
191  (g_measured_hw_rev <= 7) ? 50 :
192  100) :
193  g_measured_hw_family == 1 ?
194  150 :
195  invalid_int();
196  uint16_t idriven_hs_ma =
197  g_measured_hw_family == 0 ?
198  ((g_measured_hw_rev <= 6) ? 740 :
199  (g_measured_hw_rev <= 7) ? 100 :
200  200) :
201  g_measured_hw_family == 1 ?
202  300 :
203  invalid_int();
204 
205 
206  // Gate Drive LS Register
207  bool cbc = true; // Cycle-by cycle operation.
208  uint16_t tdrive_ns = 1000; // peak gate-current drive time
209  uint16_t idrivep_ls_ma =
210  g_measured_hw_family == 0 ?
211  ((g_measured_hw_rev <= 6) ? 370 :
212  (g_measured_hw_rev <= 7) ? 50 :
213  100) :
214  g_measured_hw_family == 1 ?
215  150 :
216  invalid_int();
217  uint16_t idriven_ls_ma =
218  g_measured_hw_family == 0 ?
219  ((g_measured_hw_rev <= 6) ? 740 :
220  (g_measured_hw_rev <= 7) ? 100 :
221  200) :
222  g_measured_hw_family == 1 ?
223  300 :
224  invalid_int();
225 
226 
227  // OCP Control Register
228  bool tretry = false; // false = 4ms, true = 50us
229  uint16_t dead_time_ns =
230  g_measured_hw_family == 0 ?
231  ((g_measured_hw_rev <= 6) ? 50 :
232  (g_measured_hw_rev <= 7) ? 200 :
233  50) :
234  g_measured_hw_family == 1 ?
235  50 :
236  invalid_int();
237  OcpMode ocp_mode = OcpMode::kLatchedFault;
238  uint8_t ocp_deg_us = 4; // valid options of 2, 4, 6, 8
239 
240  // hw rev 6 boards and later use a FET with roughly double the
241  // Rdson. We set a threshold that will trip only if we get well
242  // over the rated 100A limit.
243  uint16_t vds_lvl_mv =
244  g_measured_hw_family == 0 ?
245  ((g_measured_hw_rev <= 5) ? 260 :
246  (g_measured_hw_rev <= 7) ? 450 :
247  700) :
248  g_measured_hw_family == 1 ?
249  700 :
250  invalid_int();
251 
252 
253  // CSA Control Register
254  bool csa_fet = false;
255  bool vref_div = true; // reference voltage is VREF / 2 (bi-directional)
256  bool ls_ref = false; // false = OCP for low side is measured SHx to SPx
257  // true = OCP is measured SHx to SNx
258  uint8_t csa_gain = 20; // 5, 10, 20, 40 are options
259  bool dis_sen = false; // if true, disable sense overcurrent fault
260 
261  uint16_t sen_lvl_mv = 500; // OCP level, (250, 500, 750, 1000)
