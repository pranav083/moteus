38  void Initialize() {
39  i2c_init(&mbed_i2c_, options_.sda, options_.scl);
40  i2c_ = mbed_i2c_.i2c.handle.Instance;
41  // The mbed libraries only generate timings for a small number of
42  // fixed scenarios. We aren't in those, so we'll rely on a number
43  // from CubeMX.
44  i2c_->CR1 &= ~(I2C_CR1_PE);
45 
46  // PE must be low for a bit, so wait.
47  for (int i = 0; i < 1000; i++);
48 
49  // Now figure out the actual timing values.
50  TimingInput timing_input;
51  timing_input.peripheral_hz = HAL_RCC_GetSysClockFreq();
52  timing_input.i2c_hz = options_.frequency;
53  timing_input.i2c_mode = options_.i2c_mode;
54 
55  const auto timing = CalculateI2cTiming(timing_input);
56  if (timing.error) {
57  // These values weren't achievable. Mark everything as an
58  // error.
59  valid_ = false;
60  } else {
61  valid_ = true;
62  i2c_->CR1 =
63  (timing.digital_noise_filter << I2C_CR1_DNF_Pos) |
64  (((timing_input.analog_filter == AnalogFilter::kOff)
65  ? 1 : 0) << I2C_CR1_ANFOFF_Pos) |
66  0;
67  i2c_->TIMINGR = timing.timingr;
68  }
69 
70  // Now re-enable and wait a bit.
71  i2c_->CR1 |= (I2C_CR1_PE);
72  for (int i = 0; i < 1000; i++);
73  }
74 
75  void StartReadMemory(uint8_t slave_address,
76  uint8_t address,
77  mjlib::base::string_span data) {
78  if (!valid_) { return; }
79  if (mode_ != Mode::kIdle ||
80  (i2c_->CR2 & I2C_CR2_START) != 0 ||
81  (i2c_->ISR & I2C_ISR_BUSY) != 0) {
82  mode_ = Mode::kError;
83  return;
84  }
85 
86  slave_address_ = slave_address;
87  rx_data_ = data;
88 
89  i2c_->ICR |= (I2C_ICR_STOPCF | I2C_ICR_NACKCF);
90 
91  i2c_->CR2 = (
92  I2C_CR2_START |
93  // I2C_CR2_RD_WRN | // we are reading
94  // I2C_CR2_AUTOEND | // we are going to send a repeated start
95  (1 << I2C_CR2_NBYTES_Pos) |
96  ((slave_address_ << 1) << I2C_CR2_SADD_Pos) |
97  0);
98  i2c_->TXDR = address;
99 
100  mode_ = Mode::kSentRegister;
101  }
102 
103  enum class ReadStatus {
104  kNoStatus,
105  kComplete,
106  kError,
107  };
108 
109  ReadStatus CheckRead() {
110  if (!valid_) {
111  return ReadStatus::kError;
112  }
113 
114  if (mode_ == Mode::kComplete) {
115  mode_ = Mode::kIdle;
116  return ReadStatus::kComplete;
117  }
118  if (mode_ == Mode::kError) {
119  // Re-initialize.
120  Initialize();
121 
122  mode_ = Mode::kIdle;
123  return ReadStatus::kError;
124  }
125  return ReadStatus::kNoStatus;
126  }

128  void Poll() {
129  switch (mode_) {
130  case Mode::kIdle:
131  case Mode::kComplete:
132  case Mode::kError: {
133  break;
134  }
135  case Mode::kSentRegister: {
136  if ((i2c_->ISR & I2C_ISR_TC) == 0) {
137  break;
138  }
139 
140  // Clear any NACKs
141  i2c_->ICR |= I2C_ICR_NACKCF;
142 
143  // Now we send our repeated start to retrieve the result.
144  i2c_->CR2 =
145  (I2C_CR2_START |
146  I2C_CR2_RD_WRN |
147  I2C_CR2_AUTOEND |
148  ((slave_address_ << 1) << I2C_CR2_SADD_Pos) |
149  (rx_data_.size() << I2C_CR2_NBYTES_Pos) |
150  0);
151 
152  offset_ = 0;
153  mode_ = Mode::kReadingData;
154 
155  break;
156  }
157  case Mode::kReadingData: {
158  if ((i2c_->ISR & I2C_ISR_RXNE) == 0) {
159  break;
160  }
161 
162  // We have data.
163  rx_data_[offset_++] = i2c_->RXDR;
164 
165  if (offset_ >= rx_data_.size()) {
166  // Clear any NACKs
167  i2c_->ICR |= I2C_ICR_NACKCF;
168 
169  mode_ = Mode::kComplete;
170  }
171 
172  break;
173  }
174  }
175 
176  if (i2c_->ISR & I2C_ISR_NACKF) {
177  mode_ = Mode::kError;
178  i2c_->ICR |= I2C_ICR_NACKCF;
179  return;
180  }
181  }

183  private:
184  const Options options_;
185  bool valid_ = false;

189  kIdle,
190  kSentRegister,
191  kReadingData,

198  uint8_t slave_address_ = 0;

200  int32_t offset_ = 0;
201 };
