31 class Stm32GpioInterruptIn {
32  public:
33  using CallbackFunction = void (*)(uint32_t);
34 
35  struct Callback {
36  CallbackFunction function = {};
37  uint32_t data = 0;
38  int ref_count = 0;
39 
40  Callback() {}
41  Callback(CallbackFunction function_in, uint32_t data_in)
42  : function(function_in),
43  data(data_in) {}
44  };
45 
46  static std::optional<Stm32GpioInterruptIn>
47  Make(PinName pin, CallbackFunction function, uint32_t data) {
48  // See if something already has this interrupt channel claimed.
49  const uint32_t pin_index = STM_PIN(pin);
50  if (EXTI->IMR1 & (1 << pin_index)) {
51  // Something already has this channel. :(
52  return {};
53  }
54 
55  Stm32GpioInterruptIn result{pin, function, data};
56 
57  if (!result.entry_) {
58  // Somehow we exhausted our entry table.
59  return {};
60  }
61 
62  return result;
63  }
64

66  : pin_(rhs.pin_),
67  entry_(rhs.entry_),
68  reg_in_(rhs.reg_in_),
69  mask_(rhs.mask_) {
70  rhs.pin_ = NC;
71  }

74  pin_ = rhs.pin_;
75  entry_ = rhs.entry_;
76  reg_in_ = rhs.reg_in_;
77  mask_ = rhs.mask_;
78 
79  rhs.pin_ = NC;
80

82  }

84  Stm32GpioInterruptIn(PinName pin, CallbackFunction function, uint32_t val)
85  : pin_(pin) {
86  __HAL_RCC_SYSCFG_CLK_ENABLE();
87  __HAL_RCC_GPIOA_CLK_ENABLE();
88  __HAL_RCC_GPIOB_CLK_ENABLE();
89  __HAL_RCC_GPIOC_CLK_ENABLE();
90 
91  const uint32_t port_index = STM_PORT(pin);
92  const uint32_t pin_index = STM_PIN(pin);
93 
94  GPIO_TypeDef* gpio = reinterpret_cast<GPIO_TypeDef*>([&]() {
95  switch (port_index) {
96  case PortA: return GPIOA_BASE;
97  case PortB: return GPIOB_BASE;
98  case PortC: return GPIOC_BASE;
99  case PortD: return GPIOD_BASE;
100  case PortE: return GPIOE_BASE;
101  case PortF: return GPIOF_BASE;
102  }
103  MJ_ASSERT(false);
104  return GPIOA_BASE;
105  }());
106  reg_in_ = &gpio->IDR;
107  mask_ = static_cast<uint32_t>(1 << (static_cast<uint32_t>(pin) & 0xf));
108 
109 
110  Callback cbk(function, val);
111  entry_ = FindCallback(cbk);
112  entry_->ref_count++;
113 
114  // Set up as GPIO and input.
115  pin_function(pin, STM_PIN_DATA(STM_MODE_INPUT, GPIO_NOPULL, 0));
116 
117  // Configure our port as the one for this pin.
118  const auto shift = 4u * (pin_index & 0x03);
119  const auto original = SYSCFG->EXTICR[pin_index >> 2] & ~(0x0fu << shift);
120  SYSCFG->EXTICR[pin_index >> 2] = original | (port_index << shift);
121 
122  // We want both rising and falling edges.
123  EXTI->RTSR1 |= (1 << pin_index);
124  EXTI->FTSR1 |= (1 << pin_index);
125 
126  // Enable the external interrupt.
127  EXTI->IMR1 |= (1 << pin_index);
128 
129  const auto irq_index = FindIrqIndex(pin_);
130 
131  if (irq_count(irq_index) == 0) {
132  // We need to configure and enable this IRQ.
133  const auto irqn = FindIrqN(irq_index);
134 
135  NVIC_SetVector(
136  irqn, reinterpret_cast<uint32_t>(
137  &Stm32GpioInterruptIn::ISR_Routine));
138  HAL_NVIC_SetPriority(irqn, 1, 0);
139  NVIC_EnableIRQ(irqn);
140  }
141  (irq_count(irq_index))++;
142  }
143 
144  ~Stm32GpioInterruptIn() {
145  // Check if we're in the "unspecified but valid state"
146  // post-move-from.
147  if (pin_ == NC) { return; }
148 
149  entry_->ref_count--;
150 
151  const auto pin_index = STM_PIN(pin_);
152 
153  EXTI->IMR1 &= (1 << pin_index);
154 
155  const auto irq_index = FindIrqIndex(pin_);
156 
157  if (irq_count(irq_index) == 1) {
158  // We need to disable this IRQ.
159  const auto irqn = FindIrqN(irq_index);
160  NVIC_DisableIRQ(irqn);
161  NVIC_ClearPendingIRQ(irqn);
162  }
163  irq_count(irq_index)--;
164 
165  if (entry_->ref_count == 0) {
166  entry_->function = nullptr;
167  entry_->data = 0;
168  }
169  }
170 
171  bool read() {
172  return (*reg_in_ & mask_) != 0;
173  }
174 
175  private:
176  static Callback* FindCallback(Callback cbk) {
177  // First look for matches.
178  for (auto& entry : entries_) {
179  if (entry.function == cbk.function &&
180  entry.data == cbk.data) {
181  return &entry;
182  }
183  }
184  // Then look for empties.
185  for (auto& entry : entries_) {
186  if (entry.function == nullptr) {
187  entry = cbk;
188  return &entry;
189  }
190  }
191  return nullptr;
192  }
193 
194  static int FindIrqIndex(PinName pin) {
195  const auto pin_index = STM_PIN(pin);
196  if (pin_index <= 4) { return pin_index; }
197  if (pin_index <= 9) { return 5; }
198  if (pin_index <= 15) { return 6; }
199  return 0;
200  }
201 
202  static IRQn_Type FindIrqN(int index) {
203  constexpr IRQn_Type kIrqN[7] = {
204  EXTI0_IRQn,
205  EXTI1_IRQn,
206  EXTI2_IRQn,
207  EXTI3_IRQn,
208  EXTI4_IRQn,
209  EXTI9_5_IRQn,
210  EXTI15_10_IRQn,
211  };
212  return kIrqN[index];
213  }
214 
215  static void ISR_Routine() MOTEUS_CCM_ATTRIBUTE {
216  // Clear everything in one fell swoop!
217  EXTI->PR1 = 0x0000ffff;
218 
219  for (auto& entry : entries_) {
220  if (!entry.function) { continue; }
221  entry.function(entry.data);
222  }
223  }
224 
225  static constexpr int kMaxCallbacks = 2;
226 
227  static Callback entries_[kMaxCallbacks];
228 
229  // There are at most 7 external IRQ lines.
230  static int& irq_count(int i) {
231  static std::array<int, 7> data = { {} };
232  return data[i];
233  }
234 
235  PinName pin_;
236 
237  Callback* entry_ = nullptr;
238 
239  volatile uint32_t* reg_in_ = nullptr;
240  uint32_t mask_ = 0;
241 };
242 
243 }
