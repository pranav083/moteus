59  return static_cast<float>(static_cast<int32_t>(val)) * (1.0f / 2147483648.0f);

64  SinCos operator()(int32_t theta_q31) const {
65  SinCos result;
66  result.s = std::sin(static_cast<float>(theta_q31) * kPi * (1.0f / 2147483648.0f));
67  result.c = std::cos(static_cast<float>(theta_q31) * kPi * (1.0f / 2147483648.0f));
68  return result;
